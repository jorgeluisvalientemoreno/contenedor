SELECT P.PACKAGE_ID NUM_SOLICITUD_VENTA,
       (SELECT PT.PACKAGE_TYPE_ID||' - '||PT.DESCRIPTION 
               FROM OPEN.PS_PACKAGE_TYPE PT 
               WHERE PT.PACKAGE_TYPE_ID = P.PACKAGE_TYPE_ID) TIPO_SOLICITUD_VENTA,
       P.USER_ID DIGITADOR,
       P.PERSON_ID ID_ASESOR,
       (SELECT GP.NAME_ FROM OPEN.GE_PERSON GP WHERE GP.PERSON_ID = P.PERSON_ID) ASESOR,
       P.POS_OPER_UNIT_ID ID_PTO_ATENCION,
       (SELECT OP.NAME FROM OPEN.OR_OPERATING_UNIT OP WHERE OP.OPERATING_UNIT_ID = P.POS_OPER_UNIT_ID) PUNTO_ATENCION,
       (SELECT CT.ID_CONTRATISTA||' - '||CT.NOMBRE_CONTRATISTA 
               FROM OPEN.OR_OPERATING_UNIT OP1, OPEN.GE_CONTRATISTA CT 
               WHERE OP1.OPERATING_UNIT_ID = P.POS_OPER_UNIT_ID
               AND CT.ID_CONTRATISTA = OP1.CONTRACTOR_ID) CONTRATISTA,
        P.DOCUMENT_TYPE_ID ID_TPO_FORMU_VTA,
        (SELECT TF.TICODESC 
                FROM OPEN.TIPOCOMP TF WHERE TF.TICOCODI = P.DOCUMENT_TYPE_ID AND ROWNUM < 2) TIPO_FORMULARIO_VENTA,
        P.DOCUMENT_KEY NUMERO_FORMULARIO_VENTA,
        PR.SUBSCRIPTION_ID CONTRATO,
        PR.PRODUCT_ID PRODUCTO,
        (SELECT SE.SERVCODI||' - '||SE.SERVDESC FROM OPEN.SERVICIO SE WHERE SE.SERVCODI = PR.PRODUCT_TYPE_ID) TIPO_PRODUCTO,
        P.MOTIVE_STATUS_ID ID_ESTDO_SOL_VTA,
        (SELECT ES.DESCRIPTION FROM OPEN.PS_MOTIVE_STATUS ES WHERE ES.MOTIVE_STATUS_ID = P.MOTIVE_STATUS_ID) ESTADO_SOLICITUD_VENTA,
        P.MESSAG_DELIVERY_DATE FECHA_DIGITACION_VENTA,
        P.REQUEST_DATE FECHA_ELABORACION_VENTA,
        SB.SUBSCRIBER_ID CODIGO_CLIENTE,
        SB.IDENTIFICATION IDENTIFICACION_CLIENTE,
        SB.SUBSCRIBER_NAME||' '||SB.SUBS_LAST_NAME NOMBRE_CLIENTE,
        AB.ADDRESS DIRECCION_VENTA,
        OPEN.LDC_PK_REP_VEN.fnuCate(PR.PRODUCT_ID) ID_CATE,
        OPEN.LDC_PK_REP_VEN.fsbNombreCate(PR.PRODUCT_ID) CATEGORIA,
        OPEN.LDC_PK_REP_VEN.fnuSubCate(PR.PRODUCT_ID) ID_SUBCATE,
        OPEN.LDC_PK_REP_VEN.fsbSubCate(PR.PRODUCT_ID) SUBCATEGORIA,
        OPEN.LDC_PK_REP_VEN.fnuDepartamento(MA.GEOGRAP_LOCATION_ID) ID_DPTO,
        OPEN.LDC_PK_REP_VEN.fsbNomDepartamento(MA.GEOGRAP_LOCATION_ID) DEPARTAMENTO,
        MA.GEOGRAP_LOCATION_ID ID_LOCALIDAD,
        (SELECT L.DESCRIPTION FROM OPEN.GE_GEOGRA_LOCATION L WHERE L.GEOGRAP_LOCATION_ID = MA.GEOGRAP_LOCATION_ID) LOCALIDAD,
        OPEN.LDC_PK_REP_VEN.fnuMercaRelevanLo(MA.GEOGRAP_LOCATION_ID)||' - '||OPEN.LDC_PK_REP_VEN.fsbNombMercaRelevanLo(MA.GEOGRAP_LOCATION_ID) MERCADO_RELEVANTE,
        AB.NEIGHBORTHOOD_ID ID_BARRIO,
        (SELECT B.DESCRIPTION FROM OPEN.GE_GEOGRA_LOCATION B WHERE B.GEOGRAP_LOCATION_ID = AB.NEIGHBORTHOOD_ID) BARRIO,
        SG.OPERATING_SECTOR_ID ID_SECTOR,
        (SELECT SO.DESCRIPTION FROM OPEN.OR_OPERATING_SECTOR SO WHERE SO.OPERATING_SECTOR_ID = SG.OPERATING_SECTOR_ID) SECTOR_OPERATIVO,
        MO.COMMERCIAL_PLAN_ID ID_PLAN_COM,
        (SELECT PC.NAME FROM OPEN.CC_COMMERCIAL_PLAN PC WHERE PC.COMMERCIAL_PLAN_ID = MO.COMMERCIAL_PLAN_ID) PLAN_COMERCIAL,
        NVL((SELECT SUM(cargos.cargvalo)
                   FROM Open.cargos
                  WHERE cargos.cargdoso = 'PP-'||P.PACKAGE_ID
                    AND cargos.cargsign = 'DB'
                    AND cargos.cargtipr = 'A'
                    AND cargos.cargcaca = 53), 0) VALOR_VENTA,
         P2.PACKAGE_ID NUMERO_SOLICITUD_ANULACION,
         P2.RECEPTION_TYPE_ID ID_MEDIO_RECEP,
         (SELECT MR.DESCRIPTION FROM OPEN.GE_RECEPTION_TYPE MR WHERE MR.RECEPTION_TYPE_ID = P2.RECEPTION_TYPE_ID) MEDIO_RECEPCION_ANULACION,
         P2.USER_ID DIGITADOR_ANULACION,
         (SELECT TC.CAUSAL_TYPE_ID||' - '||TC.DESCRIPTION
                 FROM OPEN.CC_CAUSAL_TYPE TC
                 WHERE TC.CAUSAL_TYPE_ID = CC.CAUSAL_TYPE_ID) MOTIVO_REGISTRO_ANUL,
         CC.CAUSAL_ID||' - '||CC.DESCRIPTION CAUSAL_REGISTRO_ANUL,
         P2.COMMENT_ OBSERVACIONES_ANULACION,
         P2.REQUEST_DATE FECHA_REG_ANULACION,
         P2.ATTENTION_DATE FECHA_ATENCION_ANULACION,
         P2.MOTIVE_STATUS_ID ID_ESTDO_SOL_ANUL,
         (SELECT ES1.DESCRIPTION FROM OPEN.PS_MOTIVE_STATUS ES1 WHERE ES1.MOTIVE_STATUS_ID = P2.MOTIVE_STATUS_ID) ESTADO_SOLICITUD_ANULACION,
         AD.ADMIN_ACTIVITY_ID CODIGO_PROC_ADMIN,
         (SELECT WT.UNIT_TYPE_ID||' - '||WT.DESCRIPTION FROM OPEN.WF_UNIT_TYPE WT WHERE WT.UNIT_TYPE_ID = AD.UNIT_TYPE_ID) TIPO_PROC_ADMIN,
         AD.REGISTER_DATE FECHA_REGISTRO_PROC_ADMIN,
         DECODE(AD.ATTENTION_DATE,NULL,'PENDIENTE','ATENDIDO') ESTADO_PROC_ADMIN,
         (SELECT UA.PERSON_ID||' - '||UA.NAME_ 
                 FROM OPEN.MO_ACTIVITY_LOG ML, OPEN.GE_PERSON UA 
                 WHERE ML.ADMIN_ACTIVITY_ID = AD.ADMIN_ACTIVITY_ID
                 AND UA.PERSON_ID = ML.RECEIVER_PERSON_ID
                 AND ML.ACTIVITY_LOG_ID = (SELECT MAX(MLX.ACTIVITY_LOG_ID) 
                                                  FROM OPEN.MO_ACTIVITY_LOG MLX WHERE MLX.ADMIN_ACTIVITY_ID = ML.ADMIN_ACTIVITY_ID)) USUARIO_ASIGNADO,
         (SELECT AR.ORGANIZAT_AREA_ID||' - '||AR.NAME_
                 FROM OPEN.MO_ACTIVITY_LOG ML1, OPEN.GE_ORGANIZAT_AREA AR 
                 WHERE ML1.ADMIN_ACTIVITY_ID = AD.ADMIN_ACTIVITY_ID
                 AND ML1.RECEIVER_AREA_ID = AR.ORGANIZAT_AREA_ID
                 AND ML1.ACTIVITY_LOG_ID = (SELECT MAX(MLX1.ACTIVITY_LOG_ID) 
                                                   FROM OPEN.MO_ACTIVITY_LOG MLX1 WHERE MLX1.ADMIN_ACTIVITY_ID = ML1.ADMIN_ACTIVITY_ID)) AREA_ORGANIZACIONAL,
         AD.ATTENTION_DATE FECHA_ATENCION_PROC_ADMIN,
         (SELECT MM.COMMENT_TYPE_ID||' - '||MM.DESCRIPTION
                 FROM OPEN.MO_ACTIVITY_LOG ML2, OPEN.GE_COMMENT_TYPE MM 
                 WHERE ML2.ADMIN_ACTIVITY_ID = AD.ADMIN_ACTIVITY_ID
                 AND ML2.COMMENT_TYPE_ID = MM.COMMENT_TYPE_ID
                 AND ML2.ACTIVITY_LOG_ID = (SELECT MAX(MLX2.ACTIVITY_LOG_ID) 
                                                   FROM OPEN.MO_ACTIVITY_LOG MLX2 WHERE MLX2.ADMIN_ACTIVITY_ID = ML2.ADMIN_ACTIVITY_ID)) TIPO_COMENTARIO_PROC_ADMIN,
         (SELECT ML3.COMMENT_
                 FROM OPEN.MO_ACTIVITY_LOG ML3
                 WHERE ML3.ADMIN_ACTIVITY_ID = AD.ADMIN_ACTIVITY_ID
                 AND ML3.ACTIVITY_LOG_ID = (SELECT MAX(MLX3.ACTIVITY_LOG_ID) 
                                                   FROM OPEN.MO_ACTIVITY_LOG MLX3 WHERE MLX3.ADMIN_ACTIVITY_ID = ML3.ADMIN_ACTIVITY_ID)) OBSERVACIONES_PROC_ADMIN,
         (SELECT WI.INSTANCE_ID||' - '||WI.DESCRIPTION
                 FROM OPEN.WF_INSTANCE WI 
                 WHERE WI.EXTERNAL_ID = P2.PACKAGE_ID 
                 AND WI.ENTITY_ID = 17 
                 AND WI.UNIT_TYPE_ID = AD.UNIT_TYPE_ID) ETAPA_FLUJO_SOL                                               
FROM OPEN.MO_PACKAGES P,
     OPEN.MO_PACK_ANNUL_DETAIL AN,
     OPEN.MO_PACKAGES P2,
     OPEN.MO_MOTIVE MO,
     OPEN.PR_PRODUCT PR,
     OPEN.GE_SUBSCRIBER SB,
     OPEN.MO_ADDRESS MA,
     OPEN.AB_ADDRESS AB,
     OPEN.AB_SEGMENTS SG,
     OPEN.MO_MOTIVE MO2,
     OPEN.CC_CAUSAL CC,
     OPEN.MO_ADMIN_ACTIVITY AD
WHERE AN.PACKAGE_ID = P.PACKAGE_ID
AND AN.ANNUL_PACKAGE_ID = P2.PACKAGE_ID
AND MO.PACKAGE_ID(+) = P.PACKAGE_ID
AND MO.PRODUCT_TYPE_ID IN (7014,3)
AND PR.PRODUCT_ID(+) = MO.PRODUCT_ID
AND SB.SUBSCRIBER_ID(+) = P.SUBSCRIBER_ID
AND MA.PACKAGE_ID(+) = P.PACKAGE_ID
AND AB.ADDRESS_ID(+) = MA.PARSER_ADDRESS_ID
AND SG.SEGMENTS_ID(+) = AB.SEGMENT_ID
AND MO2.PACKAGE_ID(+) = P2.PACKAGE_ID 
AND CC.CAUSAL_ID(+) = MO2.CAUSAL_ID
AND AD.EXTERNAL_ID(+) = P2.PACKAGE_ID
AND P.PACKAGE_TYPE_ID IN (271,329,100229)
AND P.PACKAGE_TYPE_ID IN (SELECT TP.PACKAGE_TYPE_ID
                          FROM OPEN.PS_PACKAGE_TYPE TP
                          WHERE ','||open.dald_parameter.fsbgetvalue_chain('TIPOS_PACK_ESTADVETA')||',' LIKE '%,'||TP.PACKAGE_TYPE_ID||',%')
AND P.PACKAGE_TYPE_ID IN (271,329,100229,100271)
AND OPEN.LDC_PK_REP_VEN.fnuCate(PR.PRODUCT_ID) IN (1)
AND P2.ATTENTION_DATE BETWEEN TO_DATE(TO_CHAR('01/02/2026','dd/mm/yyyy') || '00:00:00','dd/mm/yyyy hh24:mi:ss')
                          AND TO_DATE(TO_CHAR('06/02/2026','dd/mm/yyyy') || '23:59:59','dd/mm/yyyy hh24:mi:ss')
AND	multiempresa.pkg_boconsultaempresa.fsbObtEmpresaContrato(PR.SUBSCRIPTION_ID) IN ('GDGU')
